int MOD;
out "Введите модуль, если вас не затруднит:\n";
MOD = int(in);

int fast_pow(int n, int p)
{
	if (p == 0)
		return 1;
	if (p % 2)
		return fast_pow(n, p-1) * n % MOD;
	int b = fast_pow(n, p/2);
	return b * b % MOD;
};

	{
		кабы(н % ай == 0) 
		{
			доколе(н % ай == 0)
				н /= ай;
			рез -= рез / ай;
		}
	}
	кабы(н > 1)
		рез -= рез / н;
	return рез;
};
